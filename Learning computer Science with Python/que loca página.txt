        function formatNumber(val){
            return ('0'+val).slice(-2);

        function noHayDiario(hoy) {
            var diasSinDiario = ["01-01","01-05","07-11","25-12"];
            var hoyFormatted = formatNumber(hoy.getDate())+'-'+formatNumber(hoy.getMonth()+1);

            var founded = false;
            for(var i=0, len=diasSinDiario.length; i<len && !founded; i++)
                if (hoyFormatted == diasSinDiario[i])
                    founded = true;

            return founded;
        }

        function createCLANamespace(object) {
            if(typeof(window.CLA) == 'undefined')
                window.CLA = {};

            window.CLA = objects.reduce(function(r, o){
                Object.keys(o).forEach(function (k) {
                    r[k] = o[k];
                });
                return r;
            }, {});
        }

            createCLANamespace({tapa:{}});
