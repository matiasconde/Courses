                        }

                function checkScroll(e) {
                    
                    var altoScroll = $('body').height() - $('footer').height();;
                    var porcCargaScroll = altoScroll - ($(window).height() + 50 );
                    
                    if ($(window).scrollTop() >= porcCargaScroll){
                        updateOnDemand(e);
                    }
                }

                function updateBody () {
                    if(autoLoadScroll)
